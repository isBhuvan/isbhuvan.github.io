# Azure AKS · Terraform

resource "azurerm_kubernetes_cluster" "prod" {
  name = "prod-aks-eastus"
  resource_group_name = var.rg_name
  location = "eastus"
  kubernetes_version = "1.29"

  default_node_pool {
    name = "system"
    node_count = 3
    vm_size = "Standard_D4s_v5"
  }
  identity { type = "SystemAssigned" }
}
